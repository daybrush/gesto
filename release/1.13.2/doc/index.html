Gesto

üöÄ How to use

import Gesto from "gesto";

let tx = 0;
let ty = 0;
let scale = 1;

const getso = new Gesto(target, {
    container: window,
    pinchOutside: true,
}).on("drag", e => {
    tx += e.deltaX;
    ty += e.deltaY;
    target.style.transform = `translate(${tx}px, ${ty}px) scale(${scale})`;
}).on("pinchStart", e => {
    e.datas.scale = scale;
}).on("pinch", e => {
    scale = e.datas.scale * e.scale;
    target.style.transform = `translate(${tx}px, ${ty}px) scale(${scale})`;
});

// remove event
getso.unset();
